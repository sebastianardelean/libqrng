143int main(int argc, char **argv)
144{
145
146 int opt = -1;
147 char domain_addr[DOMAIN_ADDR_LENGTH] = "\0";
148 int retval = 0;
149 int32_t *data32 = NULL;
150 int64_t *data64 = NULL;
151 float *dataf = NULL;
152 double *datad = NULL;
153 size_t i = 0;
154 FILE *stream = NULL;
155
156 /* Initialize the with the default values. */
157 strncpy(domain_addr, default_domain_addr, DOMAIN_ADDR_LENGTH);
158
159 uint32_t number_of_samples = DEFAULT_NUMBER_OF_SAMPLES;
160 double min_value_f = DEFAULT_MIN_VALUE_F;
161 double max_value_f = DEFAULT_MAX_VALUE_F;
162 int64_t min_value_i = DEFAULT_MIN_VALUE_I;
163 int64_t max_value_i = DEFAULT_MAX_VALUE_I;
164 e_rand_number_t random_number_type = NUMBER_OF_REQUESTS;
165
166
167
168 if (argc == 1) {
169 print_help();
170 exit(EXIT_FAILURE);
171
172 }
173 /* Parse the command line arguments and initialize the variables. */
174 while ((opt = getopt(argc, argv, "ha:s:m:M:i:I:t:f:")) != -1) {
175 switch (opt) {
176 case 'h':
177 print_help();
178 exit(EXIT_SUCCESS);
179 break;
180 case 'a':
181 strncpy(domain_addr, optarg, strlen(optarg));
182 break;
183 case 't':
184 random_number_type = atoi(optarg);
185 break;
186 case 's':
187 number_of_samples = atol(optarg);
188 if (number_of_samples < 1) {
189 number_of_samples = DEFAULT_NUMBER_OF_SAMPLES;
190 }
191 break;
192 case 'm':
193 min_value_f = atof(optarg);
194 break;
195 case 'M':
196 max_value_f = atof(optarg);
197 break;
198 case 'i':
199 min_value_i = atoll(optarg);
200 break;
201 case 'I':
202 max_value_i = atoll(optarg);
203 break;
204 case 'f':
205 stream = fopen(optarg, "wb");
206 break;
207 default:
208 print_help();
209 exit(EXIT_FAILURE);
210 }
211 }
212 /* If the type of request is not in range, exit with failure.*/
213 if (random_number_type >= NUMBER_OF_REQUESTS) {
214 print_help();
215 exit(EXIT_FAILURE);
216 }
217
218 /* If the ranges are not correctly defined, exit with failure.*/
219 if (min_value_f > max_value_f) {
220 print_help();
221 exit(EXIT_FAILURE);
222 }
223 if (min_value_i > max_value_i) {
224 print_help();
225 exit(EXIT_FAILURE);
226 }
227
228
229 /*Initialize qrng library*/
230 retval = qrng_open(domain_addr);
231 if (retval) {
232 exit(EXIT_FAILURE);
233 }
234
235 /* Switch on the type of request.*/
236 switch (random_number_type) {
237 case SHORT_RANDOM_NUMBER:
238 data32 = malloc(number_of_samples * sizeof(int32_t));
239 if (data32) {
240 if (qrng_random_int32(min_value_i, max_value_i, number_of_samples, data32) == 0) {
241 //print the value to stdout
242 for (i = 0; i < number_of_samples; i++) {
243 printf("%d ", data32[i]);
244 }
245 }
246
247 }
248 break;
249 case INT_RANDOM_NUMBER:
250 data64 = malloc(number_of_samples * sizeof(int64_t));
251 if (data64) {
252 if (qrng_random_int64(min_value_i, max_value_i, number_of_samples, data64) == 0) {
253 //print the value to stdout
254 for (i = 0; i < number_of_samples; i++) {
255 printf("%ld ", data64[i]);
256 }
257 }
258 }
259 break;
260 case DOUBLE_RANDOM_NUMBER:
261 datad = malloc(number_of_samples * sizeof(double));
262 if (datad) {
263 if (qrng_random_double(min_value_f, max_value_f, number_of_samples, datad) == 0) {
264 //print the value to stdout
265 for (i = 0; i < number_of_samples; i++) {
266 printf("%lf ", datad[i]);
267 }
268 }
269 }
270 break;
271 case FLOAT_RANDOM_NUMBER:
272 dataf = malloc(number_of_samples * sizeof(float));
273 if (dataf) {
274 if (qrng_random_float(min_value_f, max_value_f, number_of_samples, dataf) == 0) {
275 //print the value to stdout
276 for (i = 0; i < number_of_samples; i++) {
277 printf("%f ", dataf[i]);
278 }
279 }
280 }
281 break;
282 case STREAM_BINARY:
283 if (stream == NULL) {
284 stream = stdout;
285 }
286 (void)qrng_random_stream(stream, number_of_samples);
287 break;
288 case PERFORMANCE_REQUEST:
289 break;
290 case FIRMWARE_INFO_REQUEST:
291 (void)qrng_firmware_info(stdout);
292 break;
293 case SYSTEM_INFO_REQUEST:
294 (void)qrng_system_info(stdout);
295 break;
296 default:
297 break;
298 }
299
300 qrng_close();
301 if (data32 != NULL) {
302 free(data32);
303 }
304
305 if (data64 != NULL) {
306 free(data64);
307 }
308
309 if (dataf != NULL) {
310 free(dataf);
311 }
312
313 if (datad != NULL) {
314 free(datad);
315 }
316
317 if (stream != NULL && stream != stdout) {
318 fclose(stream);
319 }
320 exit(EXIT_SUCCESS);
321}
