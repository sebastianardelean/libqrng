99 while ((opt = getopt(argc, argv, "ha:s:f:")) != -1) {
100 switch (opt) {
101 case 'h':
102 print_help();
103 exit(EXIT_SUCCESS);
104 break;
105 case 'a':
106 strncpy(domain_addr, optarg, strlen(optarg));
107 break;
108 case 's':

110 if (number_of_samples < 1) {
111 number_of_samples = DEFAULT_NUMBER_OF_SAMPLES;
112 }
113 break;
114 case 'f':
115 stream = fopen(optarg, "wb");
116 break;
117 default:
118 print_help();
119 exit(EXIT_FAILURE);
120 }
121 }
122
123
124 /*Initialize qrng library*/
125 retval = qrng_open(domain_addr);
126 if (retval) {
127 exit(EXIT_FAILURE);
128 }
129
130
131
132 if (stream == NULL) {
133 stream = stdout;
134 }
135 (void)qrng_random_stream(stream, number_of_samples);
136
137 qrng_close();
138
139 if (stream != NULL && (stream != stdout || stream != stderr)) {
140 fclose(stream);
141 }
142 exit(EXIT_SUCCESS);
